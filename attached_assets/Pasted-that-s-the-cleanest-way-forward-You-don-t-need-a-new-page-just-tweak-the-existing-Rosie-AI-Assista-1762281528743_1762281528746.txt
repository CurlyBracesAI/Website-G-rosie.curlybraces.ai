⚙️ JS Hook (attach near the bottom of your script tag)
This sends the selected Agent to your Flask backend and updates the chat state:
document.querySelectorAll('.agent-btn').forEach(button => {
  button.addEventListener('click', async () => {
    const agent = button.dataset.agent;
    appendMessage("Rosie", `Switching to Agent ${agent}...`);

    await fetch("/set-mode", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ agent })
    });

    appendMessage("Rosie", `Okay, I’m in ${agent}-mode. What should we do first?`);
  });
});
